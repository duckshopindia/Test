📘 Method 1: OAuth Redirect (Recommended)

// JavaScript Example
const callbackUrl = encodeURIComponent(window.location.href);
const authUrl = `https://isecureid.netlify.app/?redirect_uri=${callbackUrl}`;
window.location.href = authUrl;

// Handle callback
const urlParams = new URLSearchParams(window.location.search);
if(urlParams.get('token')) {
    const user = {
        token: urlParams.get('token'),
        username: urlParams.get('username'),
        auth_id: urlParams.get('auth_id'),
        email: urlParams.get('email')
    };
    console.log('User authenticated:', user);
}

📘 Method 2: OTP API

// Send OTP via API
const response = await fetch('https://your-backend.com/send-otp', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
        'X-API-Key': 'isecure_3d69fxgwoncxwh4s0292be'
    },
    body: JSON.stringify({
        username: 'target_username',
        sender_id: 'ISECURE'
    })
});

// User receives OTP in iSecureID Messages inbox
// Verify OTP
const verifyResponse = await fetch('https://your-backend.com/verify-otp', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        username: 'target_username',
        otp: '123456'
    })
});
